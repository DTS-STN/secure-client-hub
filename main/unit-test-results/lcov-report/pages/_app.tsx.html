function aaPushEvent(errType?: string) {
  if (errType !== undefined) {
    window.adobeDataLayer?.push({
      event: 'error',
      error: {
        name: errType,
      },
    })
  } else {
    window.adobeDataLayer?.push({ event: 'pageLoad' })
  }
}
